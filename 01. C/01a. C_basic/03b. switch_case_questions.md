### Question 1:
- What is output of the following program?
```C
#include <stdio.h>

int main() {
    int n = 2;
    switch (n) {
        case 1:
            printf("1");
        case 2:
            printf("2");
        default:
            printf("d");
    }

    return 0;
}
```

### Question 7
- Explain the output of the following program
```C
#include <stdio.h>

int main() {
    int a = 7;
    int b = 10;
    int abs = 0;

    int x = a > b;
    switch (x) {
        case 1:
        {
            printf("a > b\n");
            abs = a - b;
            break;
        }

        default:
        {
            printf("a <= b\n");
            abs = b - a;
        }
        break;
    }

    printf("|a - b| = %d", abs);

    return 0;
}
```
- Output:
```
a <= b
|a - b| = 3
```
