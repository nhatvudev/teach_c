### Question 13:
- Explain output of the following program:
```C
#include <stdio.h>

int main() {
    int a, b;
    a = 1;
    b = 1;
    printf("%d && %d = %d\n", a, b, a && b);
    printf("%d || %d = %d\n\n", a, b, a || b);

    a = 1;
    b = 0;
    printf("%d && %d = %d\n", a, b, a && b);
    printf("%d || %d = %d\n\n", a, b, a || b);

    a = 0;
    b = 0;
    printf("%d && %d = %d\n", a, b, a && b);
    printf("%d || %d = %d\n", a, b, a || b);
    return 0;
}
```
Output

```
1 && 1 = 1
1 || 1 = 1

1 && 0 = 0
1 || 0 = 1

0 && 0 = 0
0 || 0 = 0
```
